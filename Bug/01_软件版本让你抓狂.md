  def modelCompile(self, met, los):
    '''model compile'''
    model = self.model
    model.compile(optimizer=self.adam,
             metrics = [met],
             loss = los)

  def modelFit(self, xTrain, yTrain, epo, batSize): 
    '''model fix'''
    model = self.model 
    model.fit(xTrain, 
              yTrain,
              epochs = epo,
              batch_size = batSize
              )
